function digPow(n, p){
    var result = 0;
    var array = [];
    var n_array = n;
    while (n_array > 0) {
        array.push(n_array - parseInt(n_array/10) * 10);
        n_array = parseInt(n_array/10);
    }
    for(var i = array.length - 1; i >= 0; i--){
        result = result + array[i] ** p;
        p++;
    }
    if (result % n === 0){
        return + result / n;
    }
    else{
        return -1;
    }
}
